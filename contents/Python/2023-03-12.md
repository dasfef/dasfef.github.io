class BusCard :
    def __init__(self) :
        self.__balance = 0

    def withDraw(self, amount) :        # 출금시 수행 함수
        self.__balance -= amount        # 금액 = 총액 - 출금액
        return self.__balance

    def deposit(self, amount) :         # 입금시 수행 함수
        self.__balance += amount        # 금액 = 총액 + 입금액
        return self.__balance
